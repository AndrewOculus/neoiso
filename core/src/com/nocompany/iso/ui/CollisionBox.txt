package com.nocompany.iso.ui;

import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.graphics.*;
import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.glutils.ShapeRenderer;


public class CollisionBox{

    private float rx, ry, sx, sy, ax, ay;
    private CollisionBoxType type;

    public CollisionBox( float rx, float ry, float sx, float sy, CollisionBoxType type ){
        this.rx = rx;
        this.ry = ry;
        this.sx = sx;
        this.sy = sy;
        this.type = type;
    }

    public boolean isTouch(float x, float y){

        float tx = x;
        float ty = y;

        if( tx > ax + rx && ty > ax + rx && tx < ax + rx + sx &&  ty < ax + ry + sy ){
            return true;
        }
        return false;
    }

    public void draw(ShapeRenderer shapeRenderer){
        shapeRenderer.rect(ax + rx, ax + ry, ax + rx + sx, ax + ry + sy);
    }

    public void setPosition(float x, float y){
        this.ax = x;
        this.ay = y;
    }

    public CollisionBoxType getType(){
        return type;
    }

}
